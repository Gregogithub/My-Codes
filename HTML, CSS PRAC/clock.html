function updateTime() { var currentTime = new Date(); var hours = currentTime.getHours(); var minutes = currentTime.getMinutes(); var seconds = currentTime.getSeconds(); // Add leading zeros if the values are less than 10 hours = (hours < 10 ? "0" : "") + hours; minutes = (minutes < 10 ? "0" : "") + minutes; seconds = (seconds < 10 ? "0" : "") + seconds; // Build the time string var timeString = hours + ":" + minutes + ":" + seconds; // Update the time display document.getElementById("time").innerHTML = timeString; } // Update the time every second setInterval(updateTime, 1000);
